One config. One report. Done.

Drop a config file, run the CLI, get HTML + JSON + JUnit + SARIF reports.

// uxinspect.config.mjs
export default {
  url: 'https://yourapp.com',
  viewports: [
    { name: 'mobile', width: 375, height: 667 },
    { name: 'desktop', width: 1280, height: 800 },
  ],
  flows: [
    {
      name: 'signup',
      steps: [
        { goto: 'https://yourapp.com' },
        { ai: 'click Get Started' },
        { fill: { selector: '#email', text: 'me@example.com' } },
        { ai: 'click Continue' },
      ],
    },
  ],
  checks: { a11y: true, perf: true, visual: true, explore: false },
  ai: { enabled: true },
  reporters: ['html', 'json', 'junit', 'sarif'],
};
